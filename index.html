// ============================================================ // LADDER // ============================================================ function renderLadder() { const me = state.players.find(p => p.id === state.currentPlayerId); const sorted = [...state.players].sort((a, b) => a.rank - b.rank); const activeChIds = new Set(state.challenges.filter(c => c.status === 'active').map(c => c.challengedId)); const meHasChallenge = me && state.challenges.some(c => c.challengerId === me.id || c.challengedId === me.id); const html = sorted.map(p => { const isMe = p.id === state.currentPlayerId; const isTarget = me && !isMe && (p.rank === me.rank - 1 || p.rank === me.rank - 2); const isChallenged = activeChIds.has(p.id); let actionHtml = ''; if (!state.currentPlayerId) { actionHtml = ``; } else if (isMe) { actionHtml = ''; } else if (isChallenged) { actionHtml = ``; } else if (isTarget && !meHasChallenge) { const targetInChallenge = state.challenges.some(c => c.challengerId === p.id || c.challengedId === p.id); actionHtml = targetInChallenge ? `` : ``; } else if (isTarget && meHasChallenge) { actionHtml = ``; } else if (p.rank >= me?.rank) { actionHtml = ''; } else { actionHtml = ``; } const rnk = p.rank <= 3 ? `r${p.rank}` : ''; const medal = p.rank === 1 ? ' 🥇' : p.rank === 2 ? ' 🥈' : p.rank === 3 ? ' 🥉' : ''; const winrate = p.wins + p.losses > 0 ? Math.round(100 * p.wins / (p.wins + p.losses)) : 0; return `
${p.rank}${medal}
${p.initials}
${p.name} ${isMe ? 'you' : ''} ${isChallenged ? `challenged by ${state.challenges.find(c => c.challengedId === p.id)?.challengerName || '?'}` : ''}
${p.wins}W · ${p.losses}L · ${winrate}% win rate
${actionHtml}
`; }).join(''); document.getElementById('tab-ladder').innerHTML = `
${html}
`; }
